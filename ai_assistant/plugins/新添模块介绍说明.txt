假设你要添加一个 语音输入插件，只需：

# plugins/voice_input.py
from plugins.plugin_base import PluginBase

class VoiceInputPlugin(PluginBase):
    def register_events(self):
        self.event_bus.subscribe('input.ready', self._on_input_ready)
    
    def name(self):
        return "VoiceInput"
    
    def _on_input_ready(self, data):
        # 处理语音输入
        pass

# main.py 中只需添加:
# from plugins.voice_input import VoiceInputPlugin
# VoiceInputPlugin(event_bus)
